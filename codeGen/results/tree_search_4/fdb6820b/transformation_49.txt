dp = [[0] * (N + 1) for _ in range(m + 1)]
for i in range(1, m + 1):
    for j in range(N + 1):
        if i <= j:
            dp[i][j] = sum(dp[k][j-k-1] for k in range(i))
        else:
            dp[i][j] = 0
dp_dict = {}
for i in range(1, m + 1):
    for j in range(N + 1):
        if dp[i][j]:
            if j not in dp_dict:
                dp_dict[j] = dp[i][j]
            else:
                dp_dict[j] += dp[i][j]
ans = dp_dict.get(N, 0)
