dp = {}

def dp_hares(n):
    if n == 1:
        return max(b[0], c[0])
    else:
        for i in range(1, n+1):
            dp[i] = max(dp[i-1][j-1], dp[i-1][j] if j < n else c[i-1]) + a[i-1]
        return dp[n][-1]

state: dp[i]: maximum joy for i hares; transition: dp[i] = max(dp[i-1][0], dp[i-1][1] if i > 1 else b[0] or c[0]) + a[i-1]
